146     /**
147      * {@inheritDoc}
148      * 
149      * @see org.apache.maven.reporting.AbstractMavenReport#executeReport(java.util.Locale)
150      * @see http://code.google.com/p/jsdoc-toolkit/wiki/CmdlineOptions
151      */
152     protected void executeReport( Locale locale )
153         throws MavenReportException
154     {
155         File workDirectory = getWorkDirectory();
156         unpackJavascriptDependency( "com.jslint:jslint", workDirectory );
157 
158         File jslint = new File( workDirectory, "fulljslint.js" );
159         List args = new ArrayList();
160         Map context = new HashMap();
161         StringBuffer options = new StringBuffer( "{" );
162         options.append( "\n  adsafe : " ).append( aDsafe ? "true" : "false" );
163         options.append( ",\n bitwise : " ).append( disallowBitwiseOperators ? "true" : "false" );
164         options.append( ",\n browser : " ).append( assumeBrowser ? "true" : "false" );
165         options.append( ",\n cap : " ).append( tolerateHtmlCase ? "true" : "false" );
166         options.append( ",\n debug : " ).append( tolerateDebuggerStatements ? "true" : "false" );
167         options.append( ",\n eqeqeq : " ).append( disallowEqualEqualandNotEqual ? "true" : "false" );
168         options.append( ",\n evil : " ).append( tolerateEval ? "true" : "false" );
169         options.append( ",\n forin : " ).append( tolerateUnfilteredForIn ? "true" : "false" );
170         options.append( ",\n fragment : " ).append( tolerateHtmlFragments ? "true" : "false" );
171         options.append( ",\n laxbreak : " ).append( tolerateSloppyLineBreaking ? "true" : "false" );
172         options.append( ",\n nomen : " ).append(
173             disallowLeadingUnderscoreInIdentifiers ? "true" : "false" );
174         options.append( ",\n on : " ).append( tolerateHtmlEventHandlers ? "true" : "false" );
175         options.append( ",\n passfail : " ).append( stopOnFirstError ? "true" : "false" );
176         options.append( ",\n plusplus : " ).append(
177             disallowPlusPlusandMinusMinus ? "true" : "false" );
178         options.append( ",\n rhino : true " );
179         options.append( ",\n undef : " ).append( disallowUndefinedVariables ? "true" : "false" );
180         options.append( ",\n white : " ).append( strictWhitespace ? "true" : "false" );
181         options.append( ",\n widget : " ).append( assumeYahooWidget ? "true" : "false" );
182         options.append( "\n}" );
183 
184         getLog().debug( "options : " + options );
185 
186         String[] scripts = getScripts( sourceDirectory );
187         Sink sink = getSink();
188         sink.head();
189         sink.title();
190         sink.text( "JsLint Javascript Verifier report" );
191         sink.title_();
192         sink.head_();
193         sink.body();
194         for ( int i = 0; i < scripts.length; i++ )
195         {
196             sink.section3();
197             sink.sectionTitle3();
198             sink.text( scripts[i] );
199             sink.sectionTitle3_();
200             sink.paragraph();
201             try
202             {
203                 String file = new File( sourceDirectory, scripts[i] ).toURL().toString();
204                 getLog().debug( "Generate jslint report for " + file );
205                 String script =
206                     "JSLINT( readUrl( '" + file + "', '" + encoding + "' ), " + options
207                         + " ); JSLINT.errors;";
208                 NativeArray errors =
209                     (NativeArray) evalScript( jslint, script, (String[]) args
210                         .toArray( new String[0] ), context );
211                 if ( errors == null || errors.getLength() == 0 )
212                 {
213                     sink.text( getBundle( locale ).getString( getName() + ".no-error" ) );
214                 }
215                 else
216                 {
217                     sink.table();
218                     sink.tableRow();
219                     sink.tableHeaderCell();
220                     sink.text( "position" );
221                     sink.tableHeaderCell_();
222                     sink.tableHeaderCell();
223                     sink.text( "error" );
224                     sink.tableHeaderCell_();
225                     sink.tableHeaderCell();
226                     sink.text( "code" );
227                     sink.tableHeaderCell_();
228                     sink.tableRow_();
229 
230                     long length = errors.getLength();
231                     for ( int j = 0; j < length; j++ )
232                     {
233                         NativeObject object = (NativeObject) errors.get( j, null );
234                         sink.tableRow();
235                         sink.tableCell();
236                         Double line = (Double) object.get( "line", null );
237                         Double character = (Double) object.get( "character", null );
238                         sink.text( line.longValue() + ":" + character.longValue() );
239                         sink.tableCell_();
240                         sink.tableCell();
241                         String reason = String.valueOf( object.get( "reason", null ) );
242                         sink.text( reason );
243                         sink.tableCell_();
244                         sink.tableCell();
245                         String evidence = String.valueOf( object.get( "evidence", null ) );
246                         sink.rawText( evidence );
247                         sink.tableCell_();
248                         sink.tableRow_();
249                     }
250                     sink.table_();
251                 }
252                 // sink.rawText( report.toString() );
253             }
254             catch ( Exception e )
255             {
256                 getLog().warn( "Error generating report for " + scripts[i], e );
257                 sink.text( "Error generating report for " + scripts[i] );
258             }
259             sink.paragraph_();
260             sink.section3_();
261         }
262         sink.body_();
263         sink.flush();
264         sink.close();
265     }
